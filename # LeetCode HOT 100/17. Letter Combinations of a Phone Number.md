/**
 * @param {string} digits
 * @return {string[]}
 */
const reflection = {
    '2': ['a', 'b', 'c'],
    '3': ['d', 'e', 'f'],
    '4': ['g', 'h', 'i'],
    '5': ['j', 'k', 'l'],
    '6': ['m', 'n', 'o'],
    '7': ['p', 'q', 'r', 's'],
    '8': ['t', 'u', 'v'],
    '9': ['w', 'x', 'y', 'z'],
}
var letterCombinations = function (digits) {
    if (digits.length == 0) {
        return []
    }
    const arr = digits.split('')
    let ans = ['']
    for (let char of arr) {
        ans = combinate(ans, reflection[char])
    }
    return ans
};

function combinate(arr1, arr2) {
    const ans = []
    for (let charA of arr1) {
        for (let charB of arr2) {
            ans.push(charA + charB)
        }
    }
    return ans
}
