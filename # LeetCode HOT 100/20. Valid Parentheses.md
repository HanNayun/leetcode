## 解析
1.遍历数组
	1.1 遇到左类括号, 入栈
	1.2遇到右类括号, 元素出栈, 将其和右类括号比对
		1.2.1 若不匹配, return false
2.若栈为空, return true; 否则return false

```JavaScript
/**
 * @param {string} s
 * @return {boolean}
 */
const left = ['(', '{', '[']
const right = [')', '}', ']']
const valid = ['()', '{}', '[]']
var isValid = function(s) {
    const stack = []
    for(let bracket of s){
        if(left.includes(bracket)){
            stack.unshift(bracket)
        }
        else if(right.includes(bracket)){
            if(!valid.includes(stack.shift()+bracket)){
                return false
            }
        }
    }
    return !(stack.length)
};
```
