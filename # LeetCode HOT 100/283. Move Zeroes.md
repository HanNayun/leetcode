```JavaScript
/**
 * @param {number[]} nums
 * @return {void} Do not return anything, modify nums in-place instead.
 */
var moveZeroes = function (nums) {
    const arr = []
    for (let i = 0; i < nums.length;) {
        if (nums[i] === 0) {
            nums.splice(i, 1)
            arr.push(0)
            continue
        }
        else {
            i++
        }
    }
    nums.splice(nums.length, 0, ...arr)
};
```
