## 解析
朴素解法, 两次遍历, 一次遍历子数组的长度, 一次遍历数组

```JavaScript
/**
 * @param {number[]} nums
 * @return {number}
 */
var maxSubArray = function (nums) {
    let width = 0
    const len = nums.length
    let maxVal = -100001
    console.log(maxVal)
    while (width++ < len) {
        let sum = 0
        let i = 0, j = 0;
        while (j < i + width) {
            sum += nums[j++]
        }
        maxVal = Math.max(sum, maxVal)

        while (j < len) {
            sum += nums[j]
            sum -= nums[i]
            maxVal = Math.max(maxVal, sum)
            j++
            i++
        }
    }
    return maxVal
};
```
