/**
 * @param {string} s
 * @return {number}
 */
var romanToInt = function (s) {
    const roma = new Map();
    roma.set('I', 1);
    roma.set('V', 5);
    roma.set('X', 10);
    roma.set('L', 50);
    roma.set('C', 100);
    roma.set('D', 500);
    roma.set('M', 1000);

    let ans = 0;
    const len = s.length;   
    let value = 0;

    for (let i = 0; i < len; i++) 
    {
        value = roma.get(s[i]);
        if(i<len-1 && value<roma.get(s[i+1]))
        {
            ans-=value;
        }
        else
        {
            ans+=value;
        }
    }
    return ans;
};
