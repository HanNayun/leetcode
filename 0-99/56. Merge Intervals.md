/**
 * @param {number[][]} intervals
 * @return {number[][]}
 */
var merge = function (intervals) {
    const stack = []
    let right = -1
    intervals.sort((a, b) => a[0] != b[0] ? a[0] - b[0] : a[1] - b[1])

    for (let [l, r] of intervals) {
        if (l <= right) {
            right = Math.max(r, right)
            stack[0][1] = right
        } else {
            stack.unshift([l, r])
            right = r
        }
    }
    return stack.reverse()
};
