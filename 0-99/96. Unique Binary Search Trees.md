/**
 * @param {number} n
 * @return {number}
 */

var numTrees = function (n) {
    const Gn = new Array(n + 1).fill(0)
    Gn[0] = 1
    Gn[1] = 1

    for (let i = 2; i <= n; i++) {
        for (let j = 1; j <= i; j++) {
            Gn[i] += Gn[j - 1] * Gn[i - j]
        }
    }

    return Gn[n]
};
