/**
 * @param {number[]} nums
 * @return {number}
 */
var findMin = function (nums) {
    var l = 0, r = nums.length - 1;
    while (nums[0] == nums[r]) { r--; }
    while (l < r) {
        let mid = (l + r + 1) >>> 1;
        if (nums[mid] >= nums[0]) {
            l = mid;
        }
        else {
            r = mid - 1;
        }
    }
    return nums[(l+1)%nums.length];
};
