/**
 * @param {number[]} nums
 * @param {number} k
 * @param {number} t
 * @return {boolean}
 */
var containsNearbyAlmostDuplicate = function (nums, k, t) {
    const len = nums.length;
    for (let width = 1; width <= k; width++) {
        let l = 0;
        let r = l + width;
        while (r < len) {
            if (Math.abs(nums[l] - nums[r]) <= t) {
                return true;
            }
            r++;
            l++;
        }
    }
    return false;
};
