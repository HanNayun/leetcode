/**
 * @param {number[]} nums1
 * @param {number[]} nums2
 * @return {number}
 */
var findMedianSortedArrays = function (nums1, nums2) {
    const len1 = nums1.length;
    const len2 = nums2.length;

    if ((len1 + len2) % 2 == 1) {
        return findMedian(nums1, nums2, ((len1 + len2) >>> 1) + 1);
    }
    else {
        return (findMedian(nums1, nums2, ((len1 + len2) >>> 1)) + findMedian(nums1, nums2, ((len1 + len2) >>> 1) + 1)) / 2;
    }
};

function findMedian(nums1, nums2, k) {
    const len1 = nums1.length;
    const len2 = nums2.length;
    var index1 = 0;
    var index2 = 0;

    while (true) {
        //Special case
        if (index1 == len1) { return nums2[index2 + k - 1] }
        if (index2 == len2) { return nums1[index1 + k - 1] }
        if (k == 1) { return nums1[index1] <= nums2[index2] ? nums1[index1] : nums2[index2]; }

        //Noraml case
        let half = (k >>> 1);
        let newIndex1 = Math.min(index1 + half, len1)-1;
        let newIndex2 = Math.min(index2 + half, len2)-1;

        if (nums1[newIndex1] <= nums2[newIndex2]) {
            k -= (newIndex1 - index1 + 1);
            index1 = newIndex1 + 1;
        }
        else {
            k -= (newIndex2 - index2 + 1);
            index2 = newIndex2 + 1;
        }
    }
}
