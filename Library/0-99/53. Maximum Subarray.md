## 解析
朴素解法, 两次遍历, 一次遍历子数组的长度, 一次遍历数组

```JavaScript
/**
 * @param {number[]} nums
 * @return {number}
 */
var maxSubArray = function (nums) {
    return dynamicProgram(nums)
};

function slideWindow(nums){
    let 
        width = 0,
        maxVal = -100001
    const len = nums.length

    while (width++ < len) {
        let sum = 0
        let i = 0, j = 0;
        while (j < i + width) {
            sum += nums[j++]
        }
        maxVal = Math.max(sum, maxVal)

        while (j < len) {
            sum += nums[j]
            sum -= nums[i]
            maxVal = Math.max(maxVal, sum)
            j++
            i++
        }
    }
    return maxVal
}

function dynamicProgram(nums){
    const
        len = nums.length,
        dp = new Array(len).fill(0)

    dp[0] = nums[0]
    for (let i = 1; i < len; i++) {
        dp[i] = Math.max(dp[i - 1] + nums[i], nums[i])
    }

    return Math.max(...dp)
}

function dynamicProgramOptimise(nums){
    let
        pre = 0,
        maxVal = -Infinity

    for (let i of nums) {
        pre += i
        pre = Math.max(pre, i)
        maxVal = Math.max(maxVal, pre)
    }
    
    return maxVal
}
```
