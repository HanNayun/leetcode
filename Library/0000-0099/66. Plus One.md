/**
 * @param {number[]} digits
 * @return {number[]}
 */
var plusOne = function (digits) {
    const
        n = digits.length

    let
        up = 1

    for (let i = n - 1; i >= 0; i--) {
        let n = digits[i] + up
        if (n <= 9) {
            digits[i] = n
            return digits
        } else {
            up = 1
            digits[i] = 0
            if (i == 0) {
                digits.unshift(1)
                return digits
            }
        }
    }

};
