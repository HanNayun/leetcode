/**
 * @param {string} a
 * @param {string} b
 * @return {string}
 */
var addBinary = function (a, b) {
    a = a.split('').reverse()
    b = b.split('').reverse()
    console.log(a, b)
    const
        la = a.length,
        lb = b.length,
        n = Math.max(a.length, b.length),
        res = new Array(n + 1).fill(0)
    let up = 0
    for (let i = 0; i <= n; i++) {
        let sum =
            up
            + (i >= la ? 0 : Number(a[i]))
            + (i >= lb ? 0 : Number(b[i]))
        if (sum >= 2) {
            up = 1
            sum %= 2
        } else {
            up = 0
        }
        res[i] = sum
    }
    if (res[n] == '0') {
        res.pop()
    }
    return res.reverse().join("")
};
