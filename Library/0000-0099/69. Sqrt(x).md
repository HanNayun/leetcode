/**
 * @param {number} x
 * @return {number}
 */
var mySqrt = function (x) {
    let
        res = 0,
        left = 0,
        right = x

    while (left <= right) {
        const mid = ((right - left) >>> 1) + left
        if (mid*mid <= x) {
            res = mid
            left = mid + 1
        } else {
            right = mid - 1
        }
    }
    return res
};
