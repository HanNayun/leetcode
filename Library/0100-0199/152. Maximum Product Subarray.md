##　Linked Problems
[53. Maximum Subarray](../0-99/53.%20Maximum%20Subarray.md)

```JavaScript
/**
 * @param {number[]} nums
 * @return {number}
 */
var maxProduct = function (nums) {
    const
        minF = [...nums],
        maxF = [...nums],
        n = nums.length

    for (let i = 1; i < n; i++) {
        minF[i] = Math.min(minF[i - 1] * nums[i], maxF[i - 1] * nums[i], nums[i])
        maxF[i] = Math.max(minF[i - 1] * nums[i], maxF[i - 1] * nums[i], nums[i])
    }

    return Math.max(...maxF)
};
```
