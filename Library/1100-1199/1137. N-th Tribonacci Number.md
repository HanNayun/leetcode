```JavaScript
/**
 * @param {number} n
 * @return {number}
 */
const T = new Map()
T.set(0, 0)
T.set(1, 1)
T.set(2, 1)
var tribonacci = function (n) {
    if (T.has(n)) {
        return T.get(n)
    }

    for (let i = 3; i <= n; i++) {
        if (T.has(i)) {
            continue
        } else {
            T.set(i, T.get(i - 1) + T.get(i - 2) + T.get(i - 3))
        }
    }

    return T.get(n)
};
```
