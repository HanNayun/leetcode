```JavaScript
/**
 * @param {number} n
 * @return {number}
 */
const F = new Map()
F.set(0, 0)
F.set(1, 1)
var fib = function (n) {
    if (F.has(n)) {
        return F.get(n)
    }

    for (let i = 2; i <= n; i++) {
        if (!F.has(i)) {
            F.set(i, F.get(i - 1) + F.get(i - 2))
        }
    }
    
    return F.get(n)
};
```
