function BinarySearch(arr, left, right, target) {
    while (left <= right) {
        let mid = left + ((right - left) >> 1);
        if (arr[mid] == target) { return mid; }
        else if (arr[mid] > target) { right = mid - 1; }
        else if (arr[mid] < target) { left = mid + 1; }
    }
    return -1;
}
