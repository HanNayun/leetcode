/**
 * @param {string} s
 * @return {string}
 */
var removeDuplicates = function (s) {
    s = s.split("");
    for (let ans = remove_once(s); ans.flag; ans = remove_once(s)) {
        s = ans.str;
    }
    return s.join("");
}

function remove_once(s) {
    let flag = false;
    for (let i = 0; i < s.length - 1;) {
        if (s[i] == s[i + 1]) {
            s.splice(i, 2);
            flag = true;
        }
        else {
            i++;
        }
    }
    return {
        'str': s,
        'flag': flag
    };
}
