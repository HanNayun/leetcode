var romanToInt = function (s) {
    const roma = new Map();
    roma.set('I', 1);
    roma.set('V', 5);
    roma.set('X', 10);
    roma.set('L', 50);
    roma.set('C', 100);
    roma.set('D', 500);
    roma.set('M', 1000);

    let ans = 0;
    const len = s.length;
    let value = 0;

    for (let i = 0; i < len; i++) {
        value = roma.get(s[i]);
        if (i < len - 1 && value < roma.get(s[i + 1])) {
            ans -= value;
        }
        else {
            ans += value;
        }
    }
    return Method_two(s);
    return ans;
};

function Method_two(s) {
    const roma = new Map();
    roma.set('I', 1); roma.set("IV", 4); roma.set('V', 5); roma.set('IX', 9);
    roma.set('X', 10); roma.set('XL', 40); roma.set('L', 50); roma.set('XC', 90); roma.set('C', 100);
    roma.set('CD', 400); roma.set('D', 500); roma.set('CM', 900); roma.set('M', 1000);

    const len = s.length;
    let ans = 0;
    for (let i = 0; i < len; i++) {
        let l = s[i];
        let r = s[i + 1];
        if (roma.get(l) < roma.get(r)) {
            ans += roma.get(r) - roma.get(l);
            i++;
        }
        else {
            ans += roma.get(l);
        }
    }
    return ans;
}
