/**
 * @param {string[]} strs
 * @return {string}
 */
var longestCommonPrefix = function (strs) {
    if (strs.length <= 1) {
        return strs[0];
    }
    return Travel_Method(strs);
};

function Travel_Method(strs) {
    let ans = "";
    const len = strs.length;
    for (let i = 0; ; i++) {
        let symbol = strs[0][i];
        if (symbol === undefined) { return ans; }
        for (let str of strs) {
            if (str[i] != symbol || str == "") { return ans; }
        }
        ans += symbol;
    }
}
