/**
 * @param {number} n
 * @param {number} start
 * @return {number}
 */
var xorOperation = function (n, start) {
    var nums = new Array();
    let ans = 0;
    for (let i = 0; i < n; i++) {
        nums.push(start + 2 * i);
    }

    for (let i of nums) {
        ans ^= i;
    }
    
    return ans;
};
