Hash_Method: 利用Hash表存储出现次数, 将所有出现次数为1的数组值相加即可.

```JavaScript
/**
 * @param {number[]} nums
 * @return {number}
 */
var sumOfUnique = function (nums) {
    return Hash_Method(nums);
};

function Hash_Method(nums) {
    const hash = new Map();
    var ans =0;
    for (let i of nums) {
        if (!hash.has(i)) {
            hash.set(i, 1);
        }
        else{
            hash.set(i, hash.get(i)+1);
        }
    }

    for(let i of nums)
    {
        if(hash.get(i)==1)
        {
            ans += i;
        }
    }
    return ans;
}

function Normal_method(nums) {
    nums = nums.sort((a, b) => a - b);
    nums = remove_once(nums);
    let ans = 0;
    for (let i of nums) {
        ans += i;
    }
    return ans;
}
function remove_once(s) {
    for (let i = 0; i < s.length - 1;) {
        let j = i + 1;
        let count = 1;
        while (s[i] == s[j] && j < s.length) {
            j++;
            count++;
        }
        if (count > 1) {
            s.splice(i, count);
        }
        else {
            i++;
        }
    }
    return s;
}
```
