 * @return {string}

var longestNiceSubstring = function (s) {
    return Bit_Method(s);
};

function Travel_Method(s) {
    var ans = "";
    const len = s.length;
    for (let i = 0; i < len - 1; i++) {
        End: for (let j = i + 1; j < len; j++) {
            let sub = s.substring(i, j + 1);
            if (sub.length > ans.length) {
                for (let i of sub) {
                    if (sub.indexOf(i.toLowerCase()) == -1 || sub.indexOf(i.toUpperCase()) == -1) {
                        continue End;
                    };
                }
                ans = sub;
            }
        }
    }

    return ans;
}

function Bit_Method(s) {
    const len = s.length;
    var ans = "";
    const alphabet = "abcdefghijklmnopqrstuvwxyz"
    const ALPHABET = alphabet.toUpperCase();
    for (let i = 0; i < len; i++) {
        let upper = 0, lower = 0;
        for (let j = i; j < len; j++) {
            let low = alphabet.indexOf(s[j]);
            let up = ALPHABET.indexOf(s[j]);
            if (low != -1) {
                lower |= 1 << low;
            }
            else {
                upper |= (1 << up);
            }

            if (upper == lower && j - i + 1 > ans.length) {
                ans = s.substring(i, j + 1);
            }
        }
    }
    return ans;
}
