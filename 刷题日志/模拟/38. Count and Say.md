/**
 * @param {number} n
 * @return {string}
 */
var countAndSay = function (n) {
    const CAS = new Array(n + 1);
    CAS[1] = "1";
    for (let i = 2; i <= n; i++) {
        CAS[i] = Say(CAS[i - 1]);
    }
    return CAS[n];
};

function Say(numStr) {
    let ans = "";
    const len = numStr.length;
    for (let i = 0; i < len; i++) {
        let count = 1;
        let number = numStr[i];
        while (numStr[i + 1] == number) {
            count++;
            i++;
        }
        ans += count + "" + number + "";
    }
    return ans;
}
