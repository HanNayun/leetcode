/**
 * @param {string} num1
 * @param {string} num2
 * @return {string}
 */
var multiply = function (num1, num2) {
    if (num1 == 0 || num2 == 0) { return "0"; }
    const len_1 = num1.length;
    const len_2 = num2.length;
    let ans = new Array(len_1 + len_2).fill(0);

    for (let i = len_2 - 1; i >= 0; i--) {
        for (let j = len_1 - 1; j >= 0; j--) {
            let r = num2[i] * num1[j];
            r += ans[i + j + 1];
            ans[i + j + 1] = r % 10;
            ans[i + j] += Math.floor(r / 10);
        }
    }
    let count = 0;
    for (let i of ans) {
        if (i == 0) {
            count++;
        }
        else {
            break;
        }
    }
    ans.splice(0,count);
    return ans.join("");
};
