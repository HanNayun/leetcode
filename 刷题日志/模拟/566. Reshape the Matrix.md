/**
 * @param {number[][]} mat
 * @param {number} r
 * @param {number} c
 * @return {number[][]}
 */
var matrixReshape = function (mat, r, c) {
    const m = mat.length;
    const n = mat[0].length;
    if (r * c != m * n) { return mat; }

    const ans = new Array(r).fill(0).map(() => new Array(c).fill(0));
    let count = 0;
    for (let i = 0; i < r; i++) {
        for (let j = 0; j < c; j++) {
            ans[i][j] = mat[Math.floor(count / n)][count % n];
            count++;
        }
    }
    return ans;
};
