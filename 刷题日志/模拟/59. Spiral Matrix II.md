/**
 * @param {number} n
 * @return {number[][]}
 */
var generateMatrix = function (n) {
    const Ans = new Array(n).fill(0).map(() => new Array(n).fill(0));
    let i = 0, j = 0;
    let value = 1, direction = 1;
    let width = n, height = n - 1;

    while (width + height > 0) {
        let count = 0;
        while (width > 0) {
            Ans[i][j] = value++;
            count++;
            if (count >= width) { break; }
            j += direction;
        }
        width--;
        i += direction;

        count = 0;
        while (height > 0) {
            Ans[i][j] = value++;
            count++;
            if (count >= height) { break; }
            i += direction;
        }
        height--;
        direction = 0 - direction;
        j += direction;
    }
    return Ans;
};
