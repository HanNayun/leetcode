/**
 * @param {string} s
 * @param {number} numRows
 * @return {string}
 */
var convert = function (s, numRows) {
    if (s.length == 1 || numRows == 1) {
        return s;
    }
    return Copy_Method(s, numRows);
    // return Math_Method(s, numRows);
};


function Math_Method(s, numRows) {
    const ans = [];
    const len = s.length;
    let offset = 2 * numRows - 2;
    for (let j = 0; j < numRows; j++) {
        let i = j;
        if (i == 0 || i == numRows - 1) {
            while (i < len) {
                ans.push(s[i]);
                i += offset;
            }
        }
        else {
            let pos1 = i, pos2 = 2 * numRows - i - 2;
            while (pos1 < len || pos2 < len) {
                if (pos1 < len) {
                    ans.push(s[pos1]);
                    pos1 += offset;
                }
                if (pos2 < len) {
                    ans.push(s[pos2]);
                    pos2 += offset;
                }
            }
        }
    }
    return ans.join('');
}

function Copy_Method(s, numRows) {
    const ans = new Array(numRows).fill(0).map(() => []);
    const len = s.length;
    let row = 0;
    let dirction = -1;
    for (let i = 0; i < len; i++) {
        ans[row].push(s[i]);
        if (row == 0 || row == numRows - 1) {
            dirction = 0 - dirction;
        }
        row += dirction;
    }
    return ans.map((x) => x.join('')).join('');
}
