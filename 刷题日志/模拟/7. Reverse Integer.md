/**
 * @param {number} x
 * @return {number}
 */
var reverse = function (x) {
    const nums = [];
    let ans = 0;
    let flag = 1;
    if (x < 0) {
        flag = 0 - flag;
        x *= flag;
    }

    while (x > 0) {
        nums.push(x % 10);
        x = Math.floor(x / 10);
    }
    while (nums[0] == 0) {
        nums.shift();
    }
    while (nums.length > 0) {
        ans = ans * 10 + nums.shift();
    }
    ans *= flag;
    if (ans <= Math.pow(2, 31) - 1 && ans >= -Math.pow(2, 31)) {
        return ans;
    }
    return 0;
};
