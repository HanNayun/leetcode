/**
 * @param {number[][]} matrix
 * @return {boolean}
 */
var isToeplitzMatrix = function (matrix) {
    const m = matrix.length;
    const n = matrix[0].length;
    for (let count = 0; count < n; count++) {
        let i = 0;
        let j = count;
        let sym = matrix[i][j];
        while (i < m && j < n) {
            if (matrix[i++][j++] != sym) {
                return false;
            }
        }
    }
    for (let count = 0; count < m; count++) {
        let i = count;
        let j = 0;
        let sym = matrix[i][j];
        while (i < m && j < n) {
            if (matrix[i++][j++] != sym) {
                return false;
            }
        }
    }
    return true;
};
