function Travel_Method(s) {
    var sign = 1, appear = false;
    const len = s.length, ans = [];
    let start = 0;
    //delete whitespace
    for (start = 0; s[start] == " " && start < len; start++) { };

    for (let i = start; i < len; i++) {
        if (s[i] <= 9 && s[i] >= 0 && s[i] != " ") {
            ans.push(s[i]);
            appear = true;
        }
        else if (s[i] == '-' && !appear) {
            sign = 0 - sign;
            appear = true;
        }
        else if (s[i] == '+' && !appear) {
            appear = true;
        }
        else {
            break;
        }
    }
    console.log(ans);

    for (let i = 0; ;) {
        if (ans[i] == '0') {
            ans.shift();
        }
        else {
            break;
        }
    }
    var a = ans.join("");
    a *= sign;
    if (a > Math.pow(2, 31) - 1) {
        a = Math.pow(2, 31) - 1;
    }
    else if (a < -Math.pow(2, 31)) {
        a = -Math.pow(2, 31);
    }
    return a;
}

function Automaton_Method(s) {
    let sign = 1, ans = 0;
    let state = "start";
    const Automaton = new Map();
    Automaton.set("start", ["start", "signed", "number", "end"]);
    Automaton.set("signed", ["end", "end", "number", "end"]);
    Automaton.set("number", ["end", "end", "number", "end"]);
    Automaton.set("end", ["end", "end", "end", "end"]);
    const len = s.length;
    for (let i = 0; i < len; i++) {
        state = Automaton.get(state)[getCol(s[i])];
        if (state == "end") {
            break;
        }
        if (state == "signed") {
            if (s[i] == "-") {
                sign = -1;
                continue;
            }
        }
        if (state == "number") {
            ans = (ans * 10 + Number(s[i])) + "";
        }
    }
    ans = ans * sign;
    if (ans > Math.pow(2, 31) - 1) {
        ans = Math.pow(2, 31) - 1;
    }
    else if (ans < -Math.pow(2, 31)) {
        ans = -Math.pow(2, 31);
    }
    return ans;
}
function getCol(s) {
    if (s == " ") {
        return 0;
    }
    else if (s == "+" || s == "-") {
        return 1;
    }
    else if (s <= 9 && s >= 0) {
        return 2;
    }
    else {
        return 3;
    }
}
