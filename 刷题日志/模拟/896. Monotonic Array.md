/**
 * @param {number[]} nums
 * @return {boolean}
 */
var isMonotonic = function (nums) {
    const len = nums.length;
    var flag = nums[len - 1] - nums[0];
    if (flag == 0) {
        for (let i = 0; i < len - 1; i++) {
            if (nums[i] != nums[i + 1]) {
                return false;
            }
        }
    }
    else if (flag > 0) {
        for (let i = 0; i < len - 1; i++) {
            if (nums[i] > nums[i + 1]) {
                return false;
            }
        }
    }
    else if (flag < 0) {
        for (let i = 0; i < len - 1; i++) {
            if (nums[i] < nums[i + 1]) {
                return false;
            }
        }
    }
    return true;
};
