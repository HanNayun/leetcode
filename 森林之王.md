## 问题描述
>森林里有 *n* 只动物，编号从1到 *n*，编号越小的动物能力越大。  
他们想投票选出森林之王，每只小动物可以投票选自己，或者与自己的偶像投票相同；如果没有偶像,则其崇拜动物的编号为0，只能投自己。   
已知小动物是否有偶像以及具体是谁，求每个人能够得到的最高票数是多少？

## 解析
>1.动物只会崇拜比自己更强大的动物  
 2.崇拜具有传递性, 可以将崇拜关系视为树的边
 3.注意处理原始数据
```Java
import java.util.Arrays;
import java.util.LinkedList;
import java.util.Queue;

public class Main
{

    static public void main(String[] args)
    {
        int n = 4;
        int[] animal={0, 1, 2, 1};    
        int[] ansA, ansB;
        ansA = wayA(animal, n);
        ansB = wayB(animal, n);

        for(int i:ansA)
        {
            System.out.print(i);
        }

        System.out.println();
        for(int i:ansB)
        {
            System.out.print(i);
        }

    }
    
    //从前遍历法
    static int[] wayA(int[] animal, int n)
    {
        int[] ans=new int[n];
        Queue<Integer> queue = new LinkedList<Integer>();
        Arrays.fill(ans, 1);        
        for(int i=0; i<n; i++)
        {
            queue.add(i);
            while(!queue.isEmpty())
            {
                int front = queue.poll();
                for(int j=0; j<n; j++)
                {
                    if(animal[j]==front+1 && j!=i)
                    {
                        queue.add(j);
                        ans[i]++;
                    }
                }
            }
        }

        return ans;
    }

    //从后遍历法
    static int[] wayB(int[] animal_arr, int n)
    {
        int[] animal = new int[n+1];
        int[] ans = new int[n+1];
        animal[0]=-1;
        for(int i=1; i<n+1; i++)
        {
            animal[i]=animal_arr[i-1];
        }

        for(int i=n; i>0; i--)
        {
            ans[i]++;
            ans[animal[i]] += ans[i];
        }

        return ans;
    }
}
```
